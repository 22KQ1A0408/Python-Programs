Problem 1: Sum off all prime numbers from m to n

def prime(n):
    for i in range(2,n):
        if n%i==0:
            return 0
    else:
        return 1
m=int(input())
n=int(input())
sum=0
for i in range(m,n+1):
    if prime(i)==1:
        sum=sum+i
print(sum)
