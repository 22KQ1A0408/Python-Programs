Problem 1: Reverse a number

n=int(input())
rev=0
while n>0:
    rem=n%10
    rev=(rev*10)+rem
    n=n//10
print(rev)

Problem 4: Prime numbers from m and n

def prime(n):
    for i in range(2,n):
        if n%i==0:
            return 0
    else:
        return 1
m=int(input())
n=int(input())
for i in range(m,n+1):
    if prime(i)==1:
       print(i,end=" ")
