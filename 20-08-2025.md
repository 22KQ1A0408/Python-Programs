Problem 1:Program to print the all the armstrong numbers from 1 to n

def armstrong_num(n):
    sum=0
    for i in str(n):
        sum+=int(i)**len(str(n))
    if sum==n:
        return 1
    else:
        return 0
n=int(input())
for i in range(1,n+1):
    if armstrong_num(i)==1:
        print(i,end=" ")
