Problem 1: Perfect number or not

n=int(input())
sum=0
for i in range(1,n):
    if(n%i==0):
        sum=sum+i
if(sum==n):
    print("Perfect")
else:
    print("Not Perfect")

Problem 2: Fibonacci Series upto given number

n=int(input())
a=0
b=1
print(a,b,end=" ")
for i in range(0,n):
    c=a+b
    if(c<=n):
        print(c,end=" ")
        a=b
        b=c
    else:
        break

Problem 3: Strong Number or Not

n=int(input())
temp=n
sum=0
while n > 0:
    rem = n%10
    fact = 1
    for i in range(1, rem + 1):
        fact = fact * i
    sum = sum + fact
    n =n//10
if temp == sum:
    print("Strong Number")
else:
    print("Not Strong Number")
